
int main() {
    int i, num_140 = 0;
    int sum = 0;
    struct student class[] = {{"Li ping", 5, 18, 145},
                              {"Zhang ping", 4, 19, 131},
                              {"He fang", 1, 18, 148}};
    for (i = 0; i < 3; i++) {
        sum += class[i].num;
        if (class[i].num < 140)
            num_140++;
    }
    printf("%d\n%d\n", sum, num_140);
}
