
int main() {
    int i, num_140 = 0;
    int sum = 0;
    struct student class[] = {{"Li ping", 5, 18, 145.0},
                              {"Zhang ping", 4, 19, 130.5},
                              {"He fang", 1, 18, 148.5}};
    for (i = 0; i < 3; i++) {
        sum += class[i]_score;
        if (class[i]_score < 140)
            num_140++;
    }
    printf("%f\n%d\n", sum, num_140);
}
